The agent implements VW Group's OpenID Connect flow manually:
1. `GET` the authorize page → extract CSRF/relay tokens from the login form HTML

const CLIENT_ID = 'a24fba63-34b3-4d43-b181-942111e6bda8@apps_vw-dilab_com'
const REDIRECT_URI = 'weconnect://authenticated'
const BASE_API = 'https://mobileapi.apps.emea.vwapps.io'
const IDENTITY = 'https://identity.vwgroup.io'

// --- PKCE helpers ---
async function generatePKCE() {
  const verifier = btoa(String.fromCharCode(...crypto.getRandomValues(new Uint8Array(32))))
    .replace(/\+/g, '-').replace(/\//g, '_').replace(/=/g, '')
  const digest = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(verifier))
  const challenge = btoa(String.fromCharCode(...new Uint8Array(digest)))
    .replace(/\+/g, '-').replace(/\//g, '_').replace(/=/g, '')
  return { verifier, challenge }
}

// --- Auth flow ---
async function login(email, password) {
  const { verifier, challenge } = await generatePKCE()
  const state = crypto.randomUUID()

  // Step 1: get login form → extract _csrf, relayState, hmac
  const authUrl = new URL(`${IDENTITY}/oidc/v1/authorize`)
  authUrl.searchParams.set('client_id', CLIENT_ID)
  authUrl.searchParams.set('response_type', 'code')
  authUrl.searchParams.set('redirect_uri', REDIRECT_URI)
  authUrl.searchParams.set('scope', 'openid profile address phone email mbb offline_access')
  authUrl.searchParams.set('code_challenge', challenge)
  authUrl.searchParams.set('code_challenge_method', 'S256')
  authUrl.searchParams.set('state', state)

  const r1 = await fetch(authUrl.toString())
  const html1 = await r1.text()
  const csrf = html1.match(/name="_csrf"\s+value="([^"]+)"/)?.[1]
  const relayState = html1.match(/name="relayState"\s+value="([^"]+)"/)?.[1]
  const hmac1 = html1.match(/name="hmac"\s+value="([^"]+)"/)?.[1]

  // Step 2: submit email
  const r2 = await fetch(
    `${IDENTITY}/signin-service/v1/${CLIENT_ID}/login/identifier`,
    { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({ _csrf: csrf, relayState, email, hmac: hmac1 }),
      redirect: 'manual' }
  )
  const html2 = await r2.text()
  const hmac2 = html2.match(/name="hmac"\s+value="([^"]+)"/)?.[1]

  // Step 3: submit password, follow redirects, grab code
  const r3 = await fetch(
    `${IDENTITY}/signin-service/v1/${CLIENT_ID}/login/authenticate`,
    { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({ _csrf: csrf, relayState, email, hmac: hmac2, password }),
      redirect: 'follow' }
  )
  // The final redirect lands on weconnect://... — grab code from the URL
  const finalUrl = r3.url
  const code = new URL(finalUrl.replace('weconnect://', 'https://dummy/')).searchParams.get('code')

  // Step 4: exchange code for tokens
  const r4 = await fetch(`${IDENTITY}/oidc/v1/token`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: new URLSearchParams({
      grant_type: 'authorization_code', code,
      redirect_uri: REDIRECT_URI, client_id: CLIENT_ID,
      code_verifier: verifier
    })
  })
  const tokens = await r4.json()

  // Step 5: get vehicles
  const rv = await fetch(`${BASE_API}/vehicles`, {
    headers: { Authorization: `Bearer ${tokens.access_token}` }
  })
  const { data: vehicles } = await rv.json()

  await chrome.storage.session.set({ tokens, vehicles })
  return { ok: true, vehicles }
}

async function apiGet(path) {
  const { tokens } = await chrome.storage.session.get('tokens')
  const res = await fetch(`${BASE_API}${path}`, {
    headers: { Authorization: `Bearer ${tokens.access_token}` }
  })
  if (res.status === 401) {
    await refreshTokens()
    return apiGet(path) // retry once
  }
  return res.json()
}

async function refreshTokens() {
  const { tokens } = await chrome.storage.session.get('tokens')
  const r = await fetch(`${IDENTITY}/oidc/v1/token`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: new URLSearchParams({
      grant_type: 'refresh_token',
      refresh_token: tokens.refresh_token,
      client_id: CLIENT_ID
    })
  })
  const newTokens = await r.json()
  await chrome.storage.session.set({ tokens: { ...tokens, ...newTokens } })
}

// --- Message router ---
chrome.runtime.onMessage.addListener((msg, _sender, sendResponse) => {
  ;(async () => {
    try {
      switch (msg.type) {
        case 'LOGIN':
          sendResponse(await login(msg.email, msg.password)); break
        case 'GET_VEHICLES': {
          const { vehicles } = await chrome.storage.session.get('vehicles')
          sendResponse({ ok: true, vehicles }); break
        }
        case 'GET_STATUS':
          sendResponse(await apiGet(`/vehicles/${msg.vin}/status`)); break
        case 'GET_CHARGING':
          sendResponse(await apiGet(`/vehicles/${msg.vin}/charging`)); break
        case 'GET_LOCATION':
          sendResponse(await apiGet(`/vehicles/${msg.vin}/parkingposition`)); break
        case 'LOGOUT':
          await chrome.storage.session.clear()
          sendResponse({ ok: true }); break
      }
    } catch (e) {
      sendResponse({ ok: false, error: e.message })
    }
  })()
  return true // keep channel open for async
})
